Permutation in String

Fixed window sliding window algorithm for permutation matching

leetcode 567 | medium

算法原理

字符串排列匹配使用固定大小滑动窗口和字符频次比较，高效确定一个字符串中是否包含另一个字符串的任意排列。

算法步骤：

  1. 预处理：统计模式字符串中每个字符的频次
  2. 初始化：创建与模式字符串长度相等的固定大小窗口
  3. 滑动窗口：移动窗口遍历字符串，更新字符频次
  4. 比较频次：检查窗口频次是否与模式频次匹配
  5. 返回结果：找到匹配则返回true，否则返回false

function checkInclusion(s1, s2) {
    if (s1.length > s2.length) return false;

    // Count character frequencies in pattern
    const s1Count = {};
    for (const char of s1) {
        s1Count[char] = (s1Count[char] || 0) + 1;
    }

    const windowSize = s1.length;
    const s2Count = {};

    // Initialize window
    for (let i = 0; i < windowSize; i++) {
        const char = s2[i];
        s2Count[char] = (s2Count[char] || 0) + 1;
    }

    // Check if initial window matches
    if (matches(s1Count, s2Count)) return true;

    // Slide window
    for (let i = windowSize; i < s2.length; i++) {
        // Add new character
        const newChar = s2[i];
        s2Count[newChar] = (s2Count[newChar] || 0) + 1;

        // Remove old character
        const oldChar = s2[i - windowSize];
        s2Count[oldChar]--;
        if (s2Count[oldChar] === 0) {
            delete s2Count[oldChar];
        }

        // Check match
        if (matches(s1Count, s2Count)) return true;
    }

    return false;
}

function matches(count1, count2) {
    const keys1 = Object.keys(count1);
    const keys2 = Object.keys(count2);

    if (keys1.length !== keys2.length) return false;

    for (const key of keys1) {
        if (count1[key] !== count2[key]) return false;
    }

    return true;
}
