Minimum Window Substring

Find the minimum window in source string containing all characters of target string

leetcode 76 | hard

算法原理

最小窗口子串是滑动窗口算法的经典困难应用。在源字符串中找到包含目标字符串所有字符（包括重复字符）的最小连续子串。

算法步骤：

  1. 预处理：统计目标字符串中每个字符的频次
  2. 初始化：左右指针都从起始位置开始，创建窗口字符频次计数器
  3. 扩展窗口：右指针向前移动，将新字符加入窗口并更新频次
  4. 检查覆盖：判断当前窗口是否包含目标字符串的所有字符
  5. 收缩窗口：当窗口满足条件时，尝试移动左指针寻找更小窗口
  6. 更新最小值：记录满足条件的最小窗口
  7. 继续执行：重复步骤3-6直到右指针到达末尾

时间复杂度

O(|s| + |t|)

其中|s|和|t|分别为源字符串和目标字符串的长度

空间复杂度

O(|s| + |t|)

需要额外的哈希表存储字符频次

代码实现

function minWindow(s, t) {
    if (s.length < t.length) return "";

    // Count frequency of target string characters
    const targetFreq = new Map();
    for (const char of t) {
        targetFreq.set(char, (targetFreq.get(char) || 0) + 1);
    }

    let left = 0;
    let right = 0;
    let minLen = Infinity;
    let minStart = 0;
    let requiredChars = targetFreq.size;
    let formedChars = 0;
    const windowFreq = new Map();

    while (right < s.length) {
        // Expand window
        const rightChar = s[right];
        windowFreq.set(rightChar, (windowFreq.get(rightChar) || 0) + 1);

        // Check if frequency requirement is satisfied for a character
        if (targetFreq.has(rightChar) &&
            windowFreq.get(rightChar) === targetFreq.get(rightChar)) {
            formedChars++;
        }

        // Try to shrink window
        while (left <= right && formedChars === requiredChars) {
            const leftChar = s[left];

            // Update minimum window
            if (right - left + 1 < minLen) {
                minLen = right - left + 1;
                minStart = left;
            }

            // Remove left character
            windowFreq.set(leftChar, windowFreq.get(leftChar) - 1);
            if (targetFreq.has(leftChar) &&
                windowFreq.get(leftChar) < targetFreq.get(leftChar)) {
                formedChars--;
            }

            left++;
        }

        right++;
    }

    return minLen === Infinity ? "" : s.substring(minStart, minStart + minLen);
}
